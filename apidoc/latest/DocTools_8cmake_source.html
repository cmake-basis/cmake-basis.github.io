30 ## @brief Add documentation target.
31 #
32 # This function is used to add a software documentation files to the project
33 # which are either just copied to the installation or generated from input
34 # files such as in particular source code files and documentation files
35 # marked up using one of the supported lightweight markup languages.
36 #
37 # The supported generators are:
38 # <table border="0">
39 # <tr>
40 # @tp @b None @endtp
41 # <td>This generator simply installs the given file or all files within
42 # the specified directory.</td>
43 # </tr>
44 # <tr>
45 # @tp @b Doxygen @endtp
46 # <td>Used to generate API documentation from in-source code comments and
47 # other related files marked up using Doxygen comments. See
48 # basis_add_doxygen_doc() for more details.</td>
49 # </tr>
50 # <tr>
51 # @tp @b Sphinx @endtp
52 # <td>Used to generate documentation such as a web site from reStructuredText.
53 # See basis_add_sphinx_doc() for more details.</td>
54 # </tr>
55 # </table>
56 #
57 # @param [in] TARGET_NAME Name of the documentation target or file.
58 # @param [in] ARGN Documentation generator as "GENERATOR generator" option
59 # and additional arguments for the particular generator.
60 # The case of the generator name is ignored, i.e.,
61 # @c Doxygen, @c DOXYGEN, @c doxYgen are all valid arguments
62 # which select the @c Doxygen generator. The default generator
63 # is the @c None generator.</td>
64 #
65 # @returns Adds a custom target @p TARGET_NAME for the generation of the
66 # documentation.
67 #
68 # @sa basis_install_doc()
69 # @sa basis_add_doxygen_doc()
70 # @sa basis_add_sphinx_doc()
71 #
72 # @ingroup CMakeAPI
73 function (basis_add_doc TARGET_NAME)
74  CMAKE_PARSE_ARGUMENTS (ARGN "" "GENERATOR" "" ${ARGN})
75  if (NOT ARGN_GENERATOR)
76  set (ARGN_GENERATOR "NONE")
77  else ()
78  string (TOUPPER "${ARGN_GENERATOR}" ARGN_GENERATOR)
79  endif ()
80  if (ARGN_GENERATOR MATCHES "NONE")
81  basis_install_doc (${TARGET_NAME} ${ARGN_UNPARSED_ARGUMENTS})
82  elseif (ARGN_GENERATOR MATCHES "DOXYGEN")
83  basis_add_doxygen_doc (${TARGET_NAME} ${ARGN_UNPARSED_ARGUMENTS})
84  elseif (ARGN_GENERATOR MATCHES "SPHINX")
85  basis_add_sphinx_doc (${TARGET_NAME} ${ARGN_UNPARSED_ARGUMENTS})
86  else ()
87  message (FATAL_ERROR "Unknown documentation generator: ${ARGN_GENERATOR}.")
88  endif ()
89 endfunction ()

91 # ----------------------------------------------------------------------------
92 ## @brief Install documentation file(s).
93 #
94 # This function either adds an installation rule for a single documentation
95 # file or a directory containing multiple documentation files.
96 #
97 # Example:
98 # @code
99 # basis_install_doc ("User Manual.pdf" OUTPUT_NAME "BASIS User Manual.pdf")
100 # basis_install_doc (DeveloperManual.docx COMPONENT dev)
101 # basis_install_doc (SourceManual.html COMPONENT src)
102 # @endcode
103 #
104 # @param [in] SOURCE Documentation file or directory to install.
105 # @param [in] ARGN List of optional arguments. Valid arguments are:
106 # @par
107 # <table border="0">
108 # <tr>
109 # @tp @b COMPONENT component @endtp
110 # <td>Name of the component this documentation belongs to.
111 # Defaults to @c BASIS_RUNTIME_COMPONENT.</td>
112 # </tr>
113 # <tr>
114 # @tp @b DESTINATION dir @endtp
115 # <td>Installation directory prefix. Defaults to @c INSTALL_DOC_DIR.</td>
116 # </tr>
117 # <tr>
118 # @tp @b OUTPUT_NAME name @endtp
119 # <td>Name of file or directory after installation.</td>
120 # </tr>
121 # </table>
122 #
123 # @sa basis_add_doc()
124 function (basis_install_doc SOURCE)
125  CMAKE_PARSE_ARGUMENTS (ARGN "" "COMPONENT;DESTINATION;OUTPUT_NAME" "" ${ARGN})
126 
127  if (NOT IS_ABSOLUTE "${SOURCE}")
128  get_filename_component (SOURCE "${SOURCE}" ABSOLUTE)
129  endif ()
130  if (NOT ARGN_DESTINATION)
131  set (ARGN_DESTINATION "${INSTALL_DOC_DIR}")
132  endif ()
133  if (NOT ARGN_COMPONENT)
134  set (ARGN_COMPONENT "${BASIS_RUNTIME_COMPONENT}")
135  endif ()
136  if (NOT ARGN_COMPONENT)
137  set (ARGN_COMPONENT "Unspecified")
138  endif ()
139  if (NOT ARGN_OUTPUT_NAME)
140  basis_get_filename_component (ARGN_OUTPUT_NAME "${SOURCE}" NAME)
141  endif ()
142 
143  basis_get_relative_path (
144  RELPATH
145  "${CMAKE_SOURCE_DIR}"
146  "${CMAKE_CURRENT_SOURCE_DIR}/${ARGN_OUTPUT_NAME}"
147  )
148 
149  message (STATUS "Adding documentation ${RELPATH}...")
150 
151  if (IS_DIRECTORY "${SOURCE}")
152  basis_install_directory (
153  "${SOURCE}" "${ARGN_DESTINATION}/${ARGN_OUTPUT_NAME}"
154  COMPONENT "${ARGN_COMPONENT}"
155  )
156  else ()
157  install (
158  FILES "${SOURCE}"
159  DESTINATION "${ARGN_DESTINATION}"
160  COMPONENT "${ARGN_COMPONENT}"
161  RENAME "${ARGN_OUTPUT_NAME}"
162  )
163  endif ()
164 
165  message (STATUS "Adding documentation ${RELPATH}... - done")
166 endfunction ()

1262  endif ()
1263  elseif (NOT IS_ABSOLUTE "${SPHINX_SOURCE_DIRECTORY}")

1265  endif ()

1268  set (SPHINX_COMPONENT "${BASIS_RUNTIME_COMPONENT}")
1269  endif ()

1271  set (SPHINX_COMPONENT "Unspecified")

1281  return ()
1282  endif ()
1283  if (DEFINED Sphinx_VERSION_MAJOR AND Sphinx_VERSION_MAJOR LESS 1)
1284  if (BUILD_DOCUMENTATION)
1285  message (FATAL_ERROR "Found sphinx-build is too old (v${Sphinx_VERSION_STRING})! Please install a more recent version and/or set Sphinx-build_EXECUTABLE to a newer version or disable BUILD_DOCUMENTATION.")
1286  endif ()
1287  message (STATUS "Command sphinx-build is too old (v${Sphinx_VERSION_STRING}). Generation of ${TARGET_UID} documentation disabled.")
1288  message (STATUS "Adding documentation ${TARGET_UID}... - skipped")
1289  return ()
1290  endif ()

1298  set (SPHINX_HTML_SIDEBARS)
1299  set (SPHINX_TEMPLATES_PATH)
1300  set (SPHINX_HTML_STATIC_PATH)
1301  set (SPHINX_EXCLUDE_PATTERNS)
1302  set (SPHINX_DEPENDS)
1303  set (OPTION_NAME)
1304  set (OPTION_VALUE)
1305  set (OPTION_PATTERN "(authors?|builders?|extensions|breathe|doxylink|doxydoc|html_sidebars|templates_path|html_static_path|exclude_pattern)")
1306  foreach (ARG IN LISTS SPHINX_UNPARSED_ARGUMENTS)
1307  if (NOT OPTION_NAME OR ARG MATCHES "^[A-Z_]+$")
1308  # SPHINX_HTML_THEME_OPTIONS
1309  if (OPTION_NAME AND NOT OPTION_NAME MATCHES "^${OPTION_PATTERN}$")
1310  if (NOT OPTION_VALUE)
1311  message (FATAL_ERROR "Option ${OPTION_NAME} is missing an argument!")
1312  endif ()
1313  list (LENGTH OPTION_VALUE NUM)
1314  if (NUM GREATER 1)
1315  basis_list_to_delimited_string (OPTION_VALUE ", " NOAUTOQUOTE ${OPTION_VALUE})
1316  set (OPTION_VALUE "[${OPTION_VALUE}]")
1317  endif ()
1318  list (APPEND SPHINX_HTML_THEME_OPTIONS "'${OPTION_NAME}': ${OPTION_VALUE}")
1319  endif ()

1321  set (OPTION_NAME "${ARG}")
1322  set (OPTION_VALUE)
1323  string (TOLOWER "${OPTION_NAME}" OPTION_NAME)
1324  # BUILDER option
1325  elseif (OPTION_NAME MATCHES "^builders?$")
1326  if (ARG MATCHES "html dirhtml singlehtml pdf latex man text texinfo linkcheck")
1327  message (FATAL_ERROR "Invalid/Unsupported Sphinx builder: ${ARG}")
1328  endif ()
1329  list (APPEND SPHINX_BUILDERS "${ARG}")
1330  # AUTHORS option
1331  elseif (OPTION_NAME MATCHES "^authors?$")
1332  list (APPEND SPHINX_AUTHORS "'${ARG}'")
1333  # EXTENSIONS option
1334  elseif (OPTION_NAME MATCHES "^extensions$")
1335  # built-in extension

1345  list (APPEND SPHINX_BREATHE_TARGETS "${ARG}")

1349  list (APPEND SPHINX_BREATHE_TARGETS "${ARG}")

1351  elseif (OPTION_NAME MATCHES "^doxylink$")

1354  elseif (OPTION_NAME MATCHES "^html_sidebars$")
1355  if (NOT ARG MATCHES "\\.html?$")
1356  set (ARG "${ARG}.html")
1357  endif ()
1358  list (APPEND SPHINX_HTML_SIDEBARS "'${ARG}'")
1359  # TEMPLATES_PATH
1360  elseif (OPTION_NAME MATCHES "^templates_path$")
1361  if (NOT IS_ABSOLUTE "${ARG}")
1362  set (ARG "${SPHINX_SOURCE_DIRECTORY}/${ARG}")
1363  endif ()
1364  list (APPEND SPHINX_TEMPLATES_PATH "'${ARG}'")
1365  # HTML_STATIC_PATH
1366  elseif (OPTION_NAME MATCHES "^html_static_path$")
1367  if (NOT IS_ABSOLUTE "${ARG}")
1368  set (ARG "${SPHINX_SOURCE_DIRECTORY}/${ARG}")
1369  endif ()
1370  list (APPEND SPHINX_HTML_STATIC_PATH "'${ARG}'")

1372  elseif (OPTION_NAME MATCHES "^exclude_pattern$")
1373  list (APPEND SPHINX_EXCLUDE_PATTERNS "'${ARG}'")
1374  # value of theme option
1375  else ()
1376  if (ARG MATCHES "^(TRUE|FALSE)$")
1377  string (TOLOWER "${ARG}" "${ARG}")
1378  endif ()
1379  if (NOT ARG MATCHES "^\\[.*\\]$|^{.*}$")
1380  set (ARG "'${ARG}'")
1381  endif ()
1382  list (APPEND OPTION_VALUE "${ARG}")
1383  endif ()
1384  endforeach ()

1386  if (OPTION_NAME AND NOT OPTION_NAME MATCHES "^${OPTION_PATTERN}$")
1387  if (NOT OPTION_VALUE)
1388  message (FATAL_ERROR "Option ${OPTION_NAME} is missing an argument!")
1389  endif ()
1390  list (LENGTH OPTION_VALUE NUM)
1391  if (NUM GREATER 1)
1392  basis_list_to_delimited_string (OPTION_VALUE ", " NOAUTOQUOTE ${OPTION_VALUE})
1393  set (OPTION_VALUE "[${OPTION_VALUE}]")
1394  endif ()
1395  list (APPEND SPHINX_HTML_THEME_OPTIONS "'${OPTION_NAME}': ${OPTION_VALUE}")
1396  endif ()

1398  if (NOT SPHINX_AUTHORS)
1399  foreach (AUTHOR IN LISTS PROJECT_AUTHORS)
1400  list (APPEND SPHINX_AUTHORS "'${AUTHOR}'")
1401  endforeach ()
1402  endif ()

1404  set (SPHINX_COPYRIGHT "${PROJECT_COPYRIGHT}")
1405  endif ()
1406  # default builders
1407  if (NOT SPHINX_BUILDERS)
1408  set (SPHINX_BUILDERS html dirhtml singlehtml man pdf texinfo text linkcheck)
1409  endif ()
1410  if (SPHINX_DEFAULT_BUILDER)
1411  list (FIND SPHINX_BUILDERS "${SPHINX_DEFAULT_BUILDER}" IDX)
1412  if (IDX EQUAL -1)
1413  list (INSERT SPHINX_BUILDERS 0 "${SPHINX_DEFAULT_BUILDER}")

1416  list (GET SPHINX_BUILDERS 0 SPHINX_DEFAULT_BUILDER)
1417  endif ()

1419  if (NOT SPHINX_OUTPUT_NAME)
1420  set (SPHINX_OUTPUT_NAME "${PROJECT_NAME}")
1421  endif ()
1422  if (NOT SPHINX_OUTPUT_DIRECTORY)
1423  if (IS_ABSOLUTE "${SPHINX_OUTPUT_NAME}")
1424  get_filename_component (SPHINX_OUTPUT_DIRECTORY "${SPHINX_OUTPUT_NAME}" PATH)

1428  endif ()

1431  endif ()

1435  if (NOT IS_ABSOLUTE "${SPHINX_${B}_OUTPUT_DIRECTORY}")

1437  endif ()
1438  else ()
1439  set (SPHINX_${B}_OUTPUT_DIRECTORY "${SPHINX_OUTPUT_DIRECTORY}/${b}")
1440  endif ()
1441  endforeach ()
1442  if (IS_ABSOLUTE "${SPHINX_OUTPUT_NAME}")
1443  basis_get_relative_path (SPHINX_OUTPUT_NAME "${SPHINX_OUTPUT_DIRECTORY}" NAME_WE)
1444  endif ()

1451  endif ()
1452  if (NOT SPHINX_LATEX_MASTER_DOC)
1453  set (SPHINX_LATEX_MASTER_DOC "${SPHINX_MASTER_DOC}")

1463  endif ()
1464  if (NOT SPHINX_UNPARSED_ARGUMENTS AND SPHINX_HTML_THEME STREQUAL BASIS_SPHINX_HTML_THEME)
1465  set (SPHINX_UNPARSED_ARGUMENTS ${BASIS_SPHINX_HTML_THEME_OPTIONS})

1468  set (SPHINX_LATEX_DOCUMENT_CLASS "howto")
1469  endif ()
1470  if (NOT SPHINX_MAN_SECTION)
