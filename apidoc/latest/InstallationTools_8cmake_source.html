48 # ----------------------------------------------------------------------------
49 ## @brief Install content of source directory excluding typical files.
50 #
51 # Files which are excluded are typical backup files, system files, files
52 # from revision control systems, and CMakeLists.txt files.
53 #
54 # Example:
55 # @code
56 # basis_install_directory("${INSTALL_DATA_DIR}")
57 # basis_install_directory(. "${INSTALL_DATA_DIR}")
58 # basis_install_directory("${CMAKE_CURRENT_SOURCE_DIR}" "${INSTALL_DATA_DIR}")
59 # basis_install_directory(images "${INSTALL_DATA_DIR}/images")
60 # @endcode
61 #
62 # @param [in] ARGN The first two arguments are extracted from the beginning
63 # of this list in the named order (without option name),
64 # and the remaining arguments are passed on to CMake's
65 # <a href="http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:install">
66 # <tt>install(DIRECTORY)</tt></a> command.
67 # @par
68 # <table border="0">
69 # <tr>
70 # @tp @b SOURCE @endtp
71 # <td>Source directory. Defaults to current source directory
72 # if only one argument, the @p DESTINATION, is given./td>
73 # </tr>
74 # <tr>
75 # @tp @b DESTINATION @endtp
76 # <td>Destination directory.</td>
77 # </tr>
78 # </table>
79 #
80 # @sa http://www.cmake.org/cmake/help/cmake-2-8-docs.html#command:install
81 #
82 # @ingroup CMakeAPI
83 function (basis_install_directory)
84  if (ARGC EQUAL 1)
85  set (SOURCE "${CMAKE_CURRENT_SOURCE_DIR}")
86  set (DESTINATION "${ARGV0}")
87  set (OPTIONS "${ARGV}")
88  list (REMOVE_AT OPTIONS 0)
89  elseif (ARGC GREATER 1)
90  set (SOURCE "${ARGV0}")
91  set (DESTINATION "${ARGV1}")
92  set (OPTIONS "${ARGV}")
93  list (REMOVE_AT OPTIONS 0 1)
94  else ()
95  message (FATAL_ERROR "Too few arguments given!")
96  endif ()
97  # check arguments
98  if (NOT IS_ABSOLUTE "${SOURCE}")
99  get_filename_component (SOURCE "${SOURCE}" ABSOLUTE)
100  endif ()
101  string (REGEX REPLACE "/+$" "" SOURCE "${SOURCE}")
102  if (NOT IS_ABSOLUTE "${DESTINATION}")
103  set (DESTINATION "${CMAKE_INSTALL_PREFIX}/${DESTINATION}")
104  endif ()
105  basis_sanitize_for_regex (PROJECT_SOURCE_DIR_RE "${PROJECT_SOURCE_DIR}")
106  if ("${DESTINATION}" MATCHES "^${PROJECT_SOURCE_DIR_RE}(/|$)")
107  message (FATAL_ERROR "Installation directory ${DESTINATION} is inside the project source tree!")
108  endif ()
109  # parse options
110  set (MATCH)
111  set (EXCLUDE)
112  set (COMPONENT)
113  set (CONFIGURATIONS)
114  set (RE)
115  set (OPT)
116  set (NUM)
117  foreach (O IN LISTS OPTIONS)
118  if (O MATCHES "^(FILES_MATCHING|.*PERMISSIONS|OPTIONAL|DESTINATION|DIRECTORY)$")
119  message (FATAL_ERROR "Option ${O} not supported by basis_install_directory()!")
120  endif ()
121  if (OPT MATCHES "^(PATTERN|REGEX)$")
122  if (NUM EQUAL 1)
123  if (O MATCHES "^EXCLUDE$")
124  list (APPEND EXCLUDE "${RE}")
125  else ()
126  list (APPEND MATCH "${RE}")
127  endif ()
128  set (OPT)
129  set (RE)
130  else ()
131  if (OPT MATCHES "PATTERN")
132  string (REGEX REPLACE "(^|[^\\])\\*" "&#9734" O "${O}")
133  basis_sanitize_for_regex (O "${O}")
134  string (REPLACE "&#9734" ".*" O "${O}")
135  endif ()
136  set (RE "${O}")
137  set (NUM 1)
138  endif ()
139  elseif (OPT MATCHES "^COMPONENT$")
140  set (COMPONENT "${O}")
141  set (OPT)
142  elseif (OPT MATCHES "^CONFIGURATIONS$")
143  if (O MATCHES "^(PATTERN|REGEX|COMPONENT|CONFIGURATIONS)$")
144  set (OPT)
145  else ()
146  list (APPEND CONFIGURATIONS "${O}")
147  math (EXPR NUM "${NUM} + 1")
148  endif ()
149  endif ()
150  if (O MATCHES "^(PATTERN|REGEX|COMPONENT|CONFIGURATIONS)$")
151  if (OPT)
152  message (FATAL_ERROR "basis_install_directory(): Option ${OPT} is missing an argument!")
153  endif ()
154  set (OPT ${O})
155  set (NUM 0)
156  endif ()
157  endforeach ()
158  if (OPT MATCHES "CONFIGURATIONS" AND NOT NUM GREATER 0)
159  message (FATAL_ERROR "basis_install_directory(): Missing argument(s) for ${OPT} option!")
160  elseif (OPT MATCHES "PATTERN|REGEX")
161  if (NUM EQUAL 1)
162  list (APPEND MATCH "${RE}")
163  else ()
164  message (FATAL_ERROR "basis_install_directory(): Missing argument(s) for ${OPT} option!")
165  endif ()
166  elseif (OPT MATCHES "COMPONENT")
167  message (FATAL_ERROR "basis_install_directory(): Missing argument for ${OPT} option!")
168  endif ()
169  list (APPEND EXCLUDE "CMakeLists.txt$" "/.svn/" "/.git/" ".DS_Store$" ".*~$") # default excludes
170  basis_list_to_delimited_string (MATCH "|" NOAUTOQUOTE ${MATCH})
171  basis_list_to_delimited_string (EXCLUDE "|" NOAUTOQUOTE ${EXCLUDE})
172  string (REPLACE "\\" "\\\\" MATCH "${MATCH}")
173  string (REPLACE "\"" "\\\"" MATCH "${MATCH}")
174  string (REPLACE "\\" "\\\\" EXCLUDE "${EXCLUDE}")
175  string (REPLACE "\"" "\\\"" EXCLUDE "${EXCLUDE}")
176  # Add installation instructions. Note that install(DIRECTORY) is here not
177  # used to avoid the generation of empty directories
178  set (OPTIONS)
179  if (CONFIGURATIONS)
180  list (APPEND OPTIONS CONFIGURATIONS ${CONFIGURATIONS})
181  endif ()
182  if (COMPONENT)
183  list (APPEND OPTIONS COMPONENT ${COMPONENT})
184  endif ()
185  install (CODE
186  "# -----------------------------------------------------------------------
187  # basis_install_directory(): ${SOURCE}
188  set (BASIS_INSTALL_DIRECTORY_FILES)
189  set (BASIS_INSTALL_DIRECTORY_SOURCE \"${SOURCE}\")
190  set (BASIS_INSTALL_DIRECTORY_DESTINATION \"\$ENV{DESTDIR}${DESTINATION}\")
191  set (BASIS_INSTALL_DIRECTORY_MATCH \"${MATCH}\")
192  set (BASIS_INSTALL_DIRECTORY_EXCLUDE \"${EXCLUDE}\")
193  file (GLOB_RECURSE BASIS_INSTALL_DIRECTORY_ALL_FILES \"\${BASIS_INSTALL_DIRECTORY_SOURCE}/*\")
194  foreach (BASIS_INSTALL_DIRECTORY_FILE IN LISTS BASIS_INSTALL_DIRECTORY_ALL_FILES)
195  if (NOT BASIS_INSTALL_DIRECTORY_MATCH OR
196  BASIS_INSTALL_DIRECTORY_FILE MATCHES \"\${BASIS_INSTALL_DIRECTORY_MATCH}\" AND
197  NOT BASIS_INSTALL_DIRECTORY_FILE MATCHES \"\${BASIS_INSTALL_DIRECTORY_EXCLUDE}\")
198  list (APPEND BASIS_INSTALL_DIRECTORY_FILES \"\${BASIS_INSTALL_DIRECTORY_FILE}\")
199  endif ()
200  endforeach ()
201  foreach (BASIS_INSTALL_DIRECTORY_FILE IN LISTS BASIS_INSTALL_DIRECTORY_FILES)
202  file (RELATIVE_PATH BASIS_INSTALL_DIRECTORY_FILE \"\${BASIS_INSTALL_DIRECTORY_SOURCE}\" \"\${BASIS_INSTALL_DIRECTORY_FILE}\")
203  execute_process (
204  COMMAND \"${CMAKE_COMMAND}\" -E compare_files
205  \"\${BASIS_INSTALL_DIRECTORY_SOURCE}/\${BASIS_INSTALL_DIRECTORY_FILE}\"
206  \"\${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\"
207  RESULT_VARIABLE RC
208  OUTPUT_QUIET
209  ERROR_QUIET
210  )
211  if (RC EQUAL 0)
212  message (STATUS \"Up-to-date: \${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\")
213  else ()
214  message (STATUS \"Installing: \${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\")
215  execute_process (
216  COMMAND \"${CMAKE_COMMAND}\" -E copy_if_different
217  \"\${BASIS_INSTALL_DIRECTORY_SOURCE}/\${BASIS_INSTALL_DIRECTORY_FILE}\"
218  \"\${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\"
219  RESULT_VARIABLE RC
220  OUTPUT_QUIET
221  ERROR_QUIET
222  )
223  if (RC EQUAL 0)
224  list (APPEND CMAKE_INSTALL_MANIFEST_FILES \"\${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\")
225  else ()
226  message (STATUS \"Failed to install \${BASIS_INSTALL_DIRECTORY_DESTINATION}/\${BASIS_INSTALL_DIRECTORY_FILE}\")
227  endif ()
228  endif ()
229  endforeach ()
230  # -----------------------------------------------------------------------"
231  ${OPTIONS}
232  )
233 endfunction ()

252 # ----------------------------------------------------------------------------
253 ## @brief Add installation rule to create a symbolic link.
254 #
255 # Note that the installation rule will only be effective on a Unix-like
256 # system, i.e., one which supports the creation of a symbolic link.
257 #
258 # @param [in] OLD The value of the symbolic link.
259 # @param [in] NEW The name of the symbolic link.
260 #
261 # @returns Adds installation rule to create the symbolic link @p NEW.
262 #
263 # @ingroup CMakeAPI
264 function (basis_install_link OLD NEW)
265  set (CMD_IN
266  "
267  set (OLD \"@OLD@\")
268  set (NEW \"@NEW@\")
269 
270  if (NOT IS_ABSOLUTE \"\${OLD}\")
271  set (OLD \"\$ENV{DESTDIR}\${CMAKE_INSTALL_PREFIX}/\${OLD}\")
272  endif ()
273  if (NOT IS_ABSOLUTE \"\${NEW}\")
274  set (NEW \"\$ENV{DESTDIR}\${CMAKE_INSTALL_PREFIX}/\${NEW}\")
275  endif ()
276 
277  if (IS_SYMLINK \"\${NEW}\")
278  file (REMOVE \"\${NEW}\")
279  endif ()
280 
281  if (EXISTS \"\${NEW}\")
282  message (STATUS \"Skipping: \${NEW} -> \${OLD}\")
283  else ()
284  message (STATUS \"Installing: \${NEW} -> \${OLD}\")
285 
286  get_filename_component (SYMDIR \"\${NEW}\" PATH)
287 
288  file (RELATIVE_PATH OLD \"\${SYMDIR}\" \"\${OLD}\")
289 
290  if (NOT EXISTS \${SYMDIR})
291  file (MAKE_DIRECTORY \"\${SYMDIR}\")
292  endif ()
293 
294  execute_process (
295  COMMAND \"${CMAKE_COMMAND}\" -E create_symlink \"\${OLD}\" \"\${NEW}\"
296  RESULT_VARIABLE RETVAL
297  )
298 
299  if (NOT RETVAL EQUAL 0)
300  message (ERROR \"Failed to create (symbolic) link \${NEW} -> \${OLD}\")
301  else ()
302  list (APPEND CMAKE_INSTALL_MANIFEST_FILES \"\${NEW}\")
303  endif ()
304  endif ()
305  "
306  )
307 
308  string (CONFIGURE "${CMD_IN}" CMD @ONLY)
309  install (CODE "${CMD}")
310 endfunction ()

312 # ----------------------------------------------------------------------------
313 ## @brief Adds installation rules to create default symbolic links.
314 #
315 # This function creates for each main executable a symbolic link directly
316 # in the directory @c INSTALL_PREFIX/bin if @c INSTALL_SINFIX is TRUE and the
317 # software is installed on a Unix-like system, i.e., one which
318 # supports the creation of symbolic links.
319 #
320 # @returns Adds installation command for creation of symbolic links in the
321 # installation tree.
322 function (basis_install_links)
323  if (NOT UNIX)
324  return ()
325  endif ()
326 
327  # main executables
328  basis_get_project_property (TARGETS PROPERTY TARGETS)
329  foreach (TARGET_UID ${TARGETS})
330  get_target_property (IMPORTED ${TARGET_UID} IMPORTED)
331 
332  if (NOT IMPORTED)
333  get_target_property (BASIS_TYPE ${TARGET_UID} BASIS_TYPE)
334  get_target_property (LIBEXEC ${TARGET_UID} LIBEXEC)
335  get_target_property (IS_TEST ${TARGET_UID} TEST)
336 
337  if (BASIS_TYPE MATCHES "EXECUTABLE" AND NOT LIBEXEC AND NOT IS_TEST)
338  get_target_property (SYMLINK_NAME ${TARGET_UID} SYMLINK_NAME)
339  if (NOT "${SYMLINK_NAME}" MATCHES "^none$|^None$|^NONE$")
340  get_target_property (SYMLINK_PREFIX ${TARGET_UID} SYMLINK_PREFIX)
341  get_target_property (SYMLINK_SUFFIX ${TARGET_UID} SYMLINK_SUFFIX)
342  get_target_property (INSTALL_DIR ${TARGET_UID} RUNTIME_INSTALL_DIRECTORY)
343  if (NOT INSTALL_DIR)
344  get_target_property (INSTALL_DIR ${TARGET_UID} INSTALL_DIRECTORY)
345  endif ()
346 
347  basis_get_target_location (OUTPUT_NAME ${TARGET_UID} NAME)
348 
349  if (NOT SYMLINK_NAME)
350  set (SYMLINK_NAME "${OUTPUT_NAME}")
351  endif ()
352  if (SYMLINK_PREFIX)
353  set (SYMLINK_NAME "${SYMLINK_PREFIX}${SYMLINK_NAME}")
354  endif ()
355  if (SYMLINK_SUFFIX)
356  set (SYMLINK_NAME "${SYMLINK_NAME}${SYMLINK_SUFFIX}")
357  endif ()
358 
359  # avoid creation of symbolic link if there would be a conflict with
360  # the subdirectory in bin/ where the actual executables are installed
361  if (INSTALL_SINFIX AND "${SYMLINK_NAME}" STREQUAL "${BASIS_INSALL_SINFIX}")
362  message (STATUS \"Skipping: ${INSTALL_DIR}/${OUTPUT_NAME} -> ${INSTALL_PREFIX}/bin/${SYMLINK_NAME}\")
363  else ()
364  basis_install_link (
365  "${INSTALL_DIR}/${OUTPUT_NAME}"
366  "bin/${SYMLINK_NAME}"
367  )
368  endif ()
369  endif ()
370  endif ()
371  endif ()
372  endforeach ()
373 endfunction ()

379 # ----------------------------------------------------------------------------
380 ## @brief Register installed package with CMake.
381 #
382 # This function adds an entry to the CMake registry for packages with the
383 # path of the directory where the package configuration file is located in
384 # order to help CMake find the package.
385 #
386 # The uninstaller whose template can be found in cmake_uninstaller.cmake.in
387 # is responsible for removing the registry entry again.
388 function (basis_register_package)
389  set (PKGDIR "${CMAKE_INSTALL_PREFIX}/${INSTALL_CONFIG_DIR}")
390  set (PKGUID "${TOPLEVEL_PROJECT_PACKAGE_UID}")
391  if (WIN32)
392  install (CODE
393  "execute_process (
394  COMMAND reg add \"HKCU\\\\Software\\\\Kitware\\\\CMake\\\\Packages\\\\${PROJECT_PACKAGE_CONFIG_PREFIX}\" /v \"${PKGUID}\" /d \"${PKGDIR}\" /t REG_SZ /f
395  RESULT_VARIABLE RT
396  ERROR_VARIABLE ERR
397  OUTPUT_QUIET
398  )
399  if (RT EQUAL 0)
400  message (STATUS \"Register: Added HKEY_CURRENT_USER\\\\Software\\\\Kitware\\\\CMake\\\\Packages\\\\${PROJECT_PACKAGE_CONFIG_PREFIX}\\\\${PKGUID}\")
401  else ()
402  string (STRIP \"\${ERR}\" ERR)
403  message (STATUS \"Register: Failed to add registry entry: \${ERR}\")
404  endif ()"
405  )
406  elseif (IS_DIRECTORY "$ENV{HOME}")
407  file (WRITE "${BINARY_CONFIG_DIR}/${PROJECT_PACKAGE_CONFIG_PREFIX}RegistryFile" "${PKGDIR}")
408  install (
409  FILES "${BINARY_CONFIG_DIR}/${PROJECT_PACKAGE_CONFIG_PREFIX}RegistryFile"
410  DESTINATION "$ENV{HOME}/.cmake/packages/${PROJECT_PACKAGE_CONFIG_PREFIX}"
411  RENAME "${PKGUID}"
412  )
413  endif ()
414 endfunction ()
